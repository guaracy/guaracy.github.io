beads 1 program '100 Doors'

var
    doors : array of yesno
    color : color
    state : str
    bold  : yesno

calc main_init
    loop from:1 to:100 index:i
        loop from:i to:100 by:i index:ix
            toggle doors[ix]

grid main_draw
    horz slice
        skip 2 al
        loop reps:10
            add 20 al
            skip 2 al
    vert slice
        skip 2 al
        loop reps:10
            add 20 al
            skip 2 al
    cell
        if doors[b.cell_seq]
            color = #B9FE6F
            state = "OPEN"
            bold  = Y
        else
            color = #FFB6B5
            state = "closed"
            bold  = N
        draw_rect(fill:color, thick:1 pt)
        draw_str("{b.cell_seq}\n{state}", size:0.3, bold:bold)
